############################
#Configuration des labels #
############################
kubectl label nodes <node-name> <label-key>=<label-value>

kubectl label nodes node-1 TypeOfPod=GitLab
kubectl label nodes node-2 TypeOfPod=PostgreSQL

kubectl get nodes --show-labels

############################################
#Configuration d'un fichier de deployement #
############################################
apiVersion: v1
kind: Pod
metadata:
  name: gitlab-pod
  labels:
    env: test
spec:
  containers:
  - name: gitlab
    image: gitlab/gitlab-ce
  nodeSelector:
    TypeOfPod: GitLab
